Add to `.github/workflows/security.yml`:
```yaml
name: Security Scan
on: [pull_request]
jobs:
  scan:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - name: Security Analysis
        run: |
          docker run -v $(pwd):/target-app \
            security-automation-platform:latest
```
